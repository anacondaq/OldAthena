//Acolyte Check

function	script	Aco_check	{
		if(BaseJob == Job_Acolyte) goto JobAco;
		if(BaseJob != Job_Novice) goto NotNov;
		if(JobLevel < 10) goto JobLess10;
		return;

		JobAco:
			mes "Você se sente bem?....";
			emotion 20;
			next;
			mes "[Bispo Tomas Cerbantes]";
			mes "Você realmente já é um Noviço.....";
			next;
			mes "[Bispo Tomas Cerbantes]";
			mes "Intão, não posso fazer nada por você, até mais.";
			emotion 18;
			close;
			

		NotNov:
			mes "Sinto muito filho, você já vive para outra profissão.  Sinto muito, mas você não poderá servir a Deus.";
			close;
			

		JobLess10:
			mes "Minha criança você não possui ainda as habilidades necessárias para servir ao senhor.";
			mes "Volte por favor quando sua fé em Deus se tornar mais forte.";
			next;
			mes "[Bispo Tomas Cerbantes]";
			mes "Que deus te abençoes.";
			close;
			
}

//Archer Check

function	script	Arc_check	{

		if(Class == 4004) goto sL_Archer;
		if(BaseJob != Job_Novice) goto sL_Other;
		mes "Você esta fazendo uma ótima escolha de profissão.";
		mes "Me de por favor seu resumo de modo que eu possa verificar suas qualificações.";
		next;
		mes "(Você entrega seu resumo)";
		next;
		mes "[Guia de Arqueiro]";
		if(JobLevel < 10) goto sL_LowLvl;
		set job_archer_q,1;
		return;

		sL_Archer:
			mes "Seus espirito narulialista me diz que você já é um arqueiro....também me diz o arco e flechas em suas mãos...Haha";
			emotion 18;
			close;
			

		sL_Other:
			mes "Desculpe, você já tem uma profissão e não pode virar um arqueiro...embora eu conpreenda o porque você quer mudar de profissão."; 
			close;
			

		sL_LowLvl:
			mes "Parece que você ainda não pode se tornar um arqueiro.  Por favor olhe os requerimentos mais uma vez.";
			close;
			

}

//Swordsman Check

}
